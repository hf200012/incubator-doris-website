# bitmap_has_any

# description

# Syntax

B00LEAN BITMAP_HAS_ANY(BITMAP lhs, BITMAP rhs)

BitmapBoolean.

# example

mysql> select bitmap_has_any(to_bitmap(1),to_bitmap(2)) cnt;
+------+
| cnt  |
+------+
|    0 |
+------+

mysql> select bitmap_has_any(to_bitmap(1),to_bitmap(1)) cnt;
+------+
| cnt  |
+------+
|    1 |
+------+

# keyword

BITMAP_HAS_ANY,BITMAP
