# Get Load Info Action

# Request

GET /api/<db>/_load_info

# Description

label

# Path parameters

  • <db>

# Query parameters

  • label

    Label

# Request body

# Response

{
	"msg": "success",
	"code": 0,
	"data": {
		"dbName": "default_cluster:db1",
		"tblNames": ["tbl1"],
		"label": "my_label",
		"clusterName": "default_cluster",
		"state": "FINISHED",
		"failMsg": "",
		"trackingUrl": ""
	},
	"count": 0
